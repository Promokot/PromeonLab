# AnimatorPanelModule Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a working `AnimatorPanelModule.prefab` from scratch via a single C# Editor builder script. The builder uses Unity's native APIs (no path-based YAML mutations) and yields a prefab where all `[SerializeField]` references are pre-wired. After build, the prefab is plugged into `UserPanel.prefab._bindings` and verified at play-mode.

**Architecture:** One Editor C# class `AnimatorPanelModuleBuilder` lives in `Assets/_App/Editor/`. It exposes a static `BuildAndSave()` method (callable from `mcp__unityMCP__execute_code`) that builds the entire hierarchy in a temp GameObject, wires all components, then saves as prefab. A second static method `WireUserPanel()` updates the `UserPanel.prefab` to include the module in its NavBar bindings.

**Tech Stack:** Unity 6000.3.7f1, UGUI, TMPro, VContainer (DI runs at scene scope, not affecting the prefab build), Unity Editor APIs (`PrefabUtility`, `AssetDatabase`, `SerializedObject`).

```csharp
private static GameObject BuildRoot()
{
    var config = AssetDatabase.LoadAssetAtPath<AnimatorPanelConfig>(ConfigPath);
    if (config == null) { Debug.LogError("Missing AnimatorPanelConfig at " + ConfigPath); return null; }

    var root = new GameObject("AnimatorPanelModule",
        typeof(RectTransform), typeof(Canvas), typeof(GraphicRaycaster), typeof(Image),
        typeof(DetachablePanel), typeof(AnimatorPanelView));

    var rt = (RectTransform)root.transform;
    rt.anchorMin   = new Vector2(0.5f, 0.5f);
    rt.anchorMax   = new Vector2(0.5f, 0.5f);
    rt.pivot       = new Vector2(0.5f, 0.5f);
    rt.sizeDelta   = new Vector2(1200f, 680f);

    var bg = root.GetComponent<Image>();
    bg.color = new Color(0.10f, 0.10f, 0.10f, 1f);
    bg.raycastTarget = true;

    var canvas = root.GetComponent<Canvas>();
    canvas.overrideSorting = false;

    // EmptyStateRoot + ActiveStateRoot stubs (children built in later tasks)
    var emptyRt  = CreateChild(root.transform, "EmptyStateRoot",  typeof(RectTransform), typeof(AnimatorEmptyStateView));
    StretchFill((RectTransform)emptyRt.transform);
    emptyRt.SetActive(false);

    var activeRt = CreateChild(root.transform, "ActiveStateRoot", typeof(RectTransform), typeof(VerticalLayoutGroup));
    StretchFill((RectTransform)activeRt.transform);
    var vlg = activeRt.GetComponent<VerticalLayoutGroup>();
    vlg.padding = new RectOffset(0,0,0,0);
    vlg.spacing = 0;
    vlg.childAlignment = TextAnchor.UpperCenter;
    vlg.childControlWidth = true; vlg.childControlHeight = true;
    vlg.childForceExpandWidth = true; vlg.childForceExpandHeight = false;

    // root is hidden by default; AnimatorPanelView.Refresh() shows the correct child
    root.SetActive(false);

    return root;
}

private static GameObject CreateChild(Transform parent, string name, params System.Type[] components)
{
    var go = new GameObject(name, components);
    go.transform.SetParent(parent, false);
    return go;
}

private static void StretchFill(RectTransform rt)
{
    rt.anchorMin   = Vector2.zero;
    rt.anchorMax   = Vector2.one;
    rt.pivot       = new Vector2(0.5f, 0.5f);
    rt.anchoredPosition = Vector2.zero;
    rt.sizeDelta   = Vector2.zero;
}
```

```csharp
private static void BuildEmptyState(GameObject emptyRoot)
{
    var view = emptyRoot.GetComponent<AnimatorEmptyStateView>();

    var noSel = BuildCenteredPanel(emptyRoot.transform, "NoSelectionPanel",
        labelText: "Select an object to view its animator", labelSize: 16,
        secondaryText: null, button: null, hintText: null);

    var noCont = BuildCenteredPanel(emptyRoot.transform, "NoContainerPanel",
        labelText: "This object has no animation container yet", labelSize: 15,
        secondaryText: null,
        button: ("+ add animation", new Color(0.18f, 0.50f, 0.95f, 1f), Color.white, new Vector2(220f, 52f)),
        hintText: "creates an action container with default 60 frames @ 24 fps");

    // Wire SerializeFields
    var so = new SerializedObject(view);
    so.FindProperty("_noSelectionPanel").objectReferenceValue = noSel;
    so.FindProperty("_noContainerPanel").objectReferenceValue = noCont;
    so.FindProperty("_addAnimationButton").objectReferenceValue =
        noCont.GetComponentInChildren<Button>(includeInactive: true);
    so.ApplyModifiedPropertiesWithoutUndo();

    noSel.SetActive(true);
    noCont.SetActive(false);
}

private static GameObject BuildCenteredPanel(Transform parent, string name,
    string labelText, int labelSize, string secondaryText,
    (string text, Color bg, Color fg, Vector2 size)? button, string hintText)
{
    var go = CreateChild(parent, name, typeof(RectTransform), typeof(VerticalLayoutGroup));
    StretchFill((RectTransform)go.transform);
    var vlg = go.GetComponent<VerticalLayoutGroup>();
    vlg.padding = new RectOffset(40, 40, 60, 60);
    vlg.spacing = 16;
    vlg.childAlignment = TextAnchor.MiddleCenter;
    vlg.childControlWidth = false; vlg.childControlHeight = false;
    vlg.childForceExpandWidth = false; vlg.childForceExpandHeight = false;

    if (labelText != null)
        BuildLabel(go.transform, "Label", labelText, labelSize, new Color(0.85f, 0.85f, 0.85f, 1f));

    if (button.HasValue)
    {
        var b = button.Value;
        BuildButton(go.transform, "AddAnimationButton", b.text, b.bg, b.fg, b.size);
    }

    if (hintText != null)
        BuildLabel(go.transform, "Hint", hintText, 12, new Color(0.55f, 0.55f, 0.55f, 1f));

    return go;
}
```

```csharp
private const string UserPanelPath = "Assets/_App/Subsystems/SpatialUi/Prefabs/Panels/UserPanel/UserPanel.prefab";

[MenuItem("PromeonLab/Build/Wire UserPanel With Animator")]
public static void WireUserPanel()
{
    var modulePrefab = AssetDatabase.LoadAssetAtPath<GameObject>(PrefabPath);
    if (modulePrefab == null) { Debug.LogError("Module prefab missing — run BuildAndSave first"); return; }

    var userPanelRoot = PrefabUtility.LoadPrefabContents(UserPanelPath);
    if (userPanelRoot == null) { Debug.LogError("Failed to load UserPanel.prefab"); return; }

    try
    {
        var modulesParent = FindChildRecursive(userPanelRoot.transform, "Modules");
        if (modulesParent == null) modulesParent = userPanelRoot.transform;

        // Remove any pre-existing AnimatorPanelModule instance
        var existing = FindChildRecursive(modulesParent, "AnimatorPanelModule");
        if (existing != null) Object.DestroyImmediate(existing.gameObject);

        // Instantiate fresh
        var instance = (GameObject)PrefabUtility.InstantiatePrefab(modulePrefab, modulesParent);
        instance.name = "AnimatorPanelModule";
        instance.SetActive(false);

        // Append to _bindings
        var userPanel = userPanelRoot.GetComponent<UserPanel>();
        var so = new SerializedObject(userPanel);
        var bindings = so.FindProperty("_bindings");
        // check existing for animator entry
        int existingIdx = -1;
        for (int i = 0; i < bindings.arraySize; i++)
        {
            var elem = bindings.GetArrayElementAtIndex(i);
            var id = elem.FindPropertyRelative("EntryId").stringValue;
            if (id == "animator") { existingIdx = i; break; }
        }
        if (existingIdx < 0)
        {
            bindings.arraySize++;
            existingIdx = bindings.arraySize - 1;
        }
        var entry = bindings.GetArrayElementAtIndex(existingIdx);
        entry.FindPropertyRelative("EntryId").stringValue = "animator";
        entry.FindPropertyRelative("Panel").objectReferenceValue = instance;
        // NavButton left as-is or null — user adds manually
        so.ApplyModifiedPropertiesWithoutUndo();

        PrefabUtility.SaveAsPrefabAsset(userPanelRoot, UserPanelPath);
        Debug.Log("[AnimatorPanelModuleBuilder] UserPanel wired (binding idx " + existingIdx + ")");
    }
    finally
    {
        PrefabUtility.UnloadPrefabContents(userPanelRoot);
    }
}

private static Transform FindChildRecursive(Transform parent, string name)
{
    if (parent.name == name) return parent;
    for (int i = 0; i < parent.childCount; i++)
    {
        var r = FindChildRecursive(parent.GetChild(i), name);
        if (r != null) return r;
    }
    return null;
}
```
